columns = ['movie id','movie title','release date']
tmp.drop(columns, inplace = True,axis=1)

tmplist['Year'] = pd.DatetimeIndex(tmplist['release date']).year

for col in tmp.columns: 
    ax = tmplist.groupby('Year')[col].sum().sort_values(ascending = False)
    ax= ax.reset_index() # flatten the dataframe
